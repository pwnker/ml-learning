          &= \left(\frac{1}{n} \cdot \sum_{i=1}^{n}(x_i \cdot w - y_i)^2 \right)' = \\ 
          &= \frac{1}{n} \cdot \left(\sum_{i=1}^{n}(x_i \cdot w - y_i)^2 \right)' = \\

          &= \frac{1}{n} \cdot \sum_{i=1}^{n}\left( 2 \cdot (x_i \cdot w - y_i) \cdot x_i \right) 
\end{align}
